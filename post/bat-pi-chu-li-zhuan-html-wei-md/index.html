bat批处理转HTML为MD

2022-05-26 1 min read 
%文件重命名 去掉里面的空格 此步骤似乎可省%
@echo off&setlocal enabledelayedexpansion
for /f "delims=" %%i in ('dir /s/b *.*') do (
set "foo=%%~nxi"
set foo=!foo: =!
set foo=!foo: =!
ren "%%~fi" "!foo!"
)

%更改文件后缀%
ren *.md *.html

%用pandoc转换HTML到MD%
for %a in (*.html) do @pandoc %~na.html -o %~na.md

exit
